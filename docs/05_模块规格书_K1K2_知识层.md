# 模块规格书: K1 场景知识库 + K2 治理规则库

> 模块ID: K1-KB / K2-RULES  
> 优先级: K1=P0, K2=P1  
> 预估工时: K1 3-4天, K2 2-3天  
> 前置依赖: F1数据模型, D2数据存储  
> Claude Code 会话数: 3次

---

## 1. 模块概述

### 1.1 知识层定位

知识层(Layer 2.5)位于数据平台和Agent层之间，为Agent决策提供可配置的知识基础。

```
Agent层 (A2清洁调度)
    │
    │ 决策前查询
    ▼
知识层
├── K1 场景知识库: "应该知道什么"  (领域知识、Prompt模板、示例)
├── K2 治理规则库: "不能做什么"    (约束、偏好、策略)
└── K3 决策日志库: "做过什么"      (决策三元组 — 见B1规格书)
    │
    │ 持久化
    ▼
数据平台层 (D2 PostgreSQL + TimescaleDB)
```

### 1.2 设计原则

- **配置优于编码**: 所有领域知识和规则通过数据库配置，不硬编码在Agent中
- **条件匹配**: 知识和规则都有适用条件，Agent根据当前上下文自动匹配
- **版本审计**: 每条知识和规则都有完整的版本和审计信息
- **前瞻兼容**: 数据模型预留阶段二的层级继承和跨节点共享字段

```python
# src/knowledge/scenario_kb_service.py

from typing import List, Optional, Dict, Any, Protocol


class ScenarioKBServiceInterface(Protocol):
    """场景知识库服务接口"""
    
    # === CRUD ===
    
    async def create_knowledge(
        self, knowledge: "ScenarioKnowledge"
    ) -> str:
        """创建知识条目，返回knowledge_id"""
        ...
    
    async def get_knowledge(self, knowledge_id: str) -> Optional["ScenarioKnowledge"]:
        """获取单条知识"""
        ...
    
    async def update_knowledge(
        self, knowledge_id: str, updates: Dict[str, Any]
    ) -> bool:
        ...
    
    async def delete_knowledge(self, knowledge_id: str) -> bool:
        """软删除（设enabled=False）"""
        ...
    
    # === 查询 ===
    
    async def query_applicable_knowledge(
        self,
        scenario_category: str,
        building_type: str = "all",
        zone_id: str = "all",
        current_time: Optional[str] = None,
        conditions: Optional[Dict[str, Any]] = None,
        knowledge_types: Optional[List[str]] = None,
        max_items: int = 20
    ) -> List["ScenarioKnowledge"]:
        """
        查询当前上下文适用的知识条目
        
        匹配逻辑:
        1. enabled=True 且在有效期内
        2. scenario_category匹配
        3. applicable_building_types包含当前building_type或"all"
        4. applicable_zones包含当前zone_id或"all"
        5. applicable_time_ranges覆盖当前时间（如有配置）
        6. applicable_conditions与当前conditions匹配（如有配置）
        
        排序: priority降序
        """
        ...
    
    # === Prompt组装 ===
    
    async def get_prompt_template(
        self, template_id: str
    ) -> Optional["PromptTemplate"]:
        ...
    
    async def assemble_prompt(
        self,
        template_id: str,
        variables: Dict[str, str],
        scenario_category: str,
        context: Dict[str, Any]
    ) -> str:
        """
        组装完整的Agent Prompt
        
        流程:
        1. 获取PromptTemplate
        2. 填充variables占位符
        3. 根据knowledge_slots查询适用知识
        4. 将知识注入prompt的对应位置
        5. 检查总token数不超过max_total_tokens
        6. 返回组装后的完整prompt
        """
        ...
    
    # === 效果跟踪 ===
    
    async def record_usage(
        self, knowledge_id: str, outcome_score: Optional[float] = None
    ) -> None:
        """记录知识使用次数和效果分数"""
        ...
```

```python
# src/knowledge/rule_engine_service.py

class RuleEngineServiceInterface(Protocol):
    """治理规则引擎服务接口"""
    
    # === CRUD ===
    
    async def create_rule(self, rule: "GovernanceRule") -> str:
        """创建规则，返回rule_id"""
        ...
    
    async def get_rule(self, rule_id: str) -> Optional["GovernanceRule"]:
        ...
    
    async def update_rule(self, rule_id: str, updates: Dict[str, Any]) -> bool:
        ...
    
    async def disable_rule(self, rule_id: str) -> bool:
        """禁用规则"""
        ...
    
    # === 规则查询 ===
    
    async def get_active_rules(
        self,
        scope: Optional[str] = None,
        agent_type: Optional[str] = None,
        building_id: Optional[str] = None,
        zone_id: Optional[str] = None
    ) -> List["GovernanceRule"]:
        """
        获取当前生效的规则
        
        筛选: enabled=True, 在有效期内, 匹配scope和适用范围
        排序: priority降序
        """
        ...
    
    # === 规则评估 ===
    
    async def evaluate(
        self,
        decision: Dict[str, Any],
        context: Dict[str, Any],
        scope: str = "system",
        agent_type: str = ""
    ) -> List["RuleEvalResult"]:
        """
        对决策评估所有适用规则
        
        返回所有触发的规则评估结果（包括WARN级别）
        """
        ...
    
    async def evaluate_single(
        self,
        rule: "GovernanceRule",
        decision: Dict[str, Any],
        context: Dict[str, Any]
    ) -> "RuleEvalResult":
        """评估单条规则"""
        ...
    
    # === 规则条件引擎 ===
    
    def compile_condition(
        self, condition: Dict[str, Any]
    ) -> "CompiledCondition":
        """
        编译json-rules条件为可执行对象
        
        支持的操作符:
        - 比较: ==, !=, <, >, <=, >=
        - 逻辑: and, or, not
        - 包含: in, not_in
        - 存在: exists, not_exists
        """
        ...
```

### 3.2 条件表达式格式

```python
# json-rules兼容格式

# 简单条件
{"field": "battery_level", "operator": "<", "value": 20}

# 组合条件
{
    "and": [
        {"field": "battery_level", "operator": "<", "value": 20},
        {"field": "task_type", "operator": "!=", "value": "return_to_charger"}
    ]
}

# 嵌套条件
{
    "or": [
        {"field": "robot_status", "operator": "==", "value": "error"},
        {
            "and": [
                {"field": "battery_level", "operator": "<", "value": 5},
                {"field": "distance_to_charger", "operator": ">", "value": 100}
            ]
        }
    ]
}

# 上下文字段访问（使用点号路径）
{"field": "environmental_state.weather", "operator": "==", "value": "rainy"}
{"field": "robot_states.robot-001.battery_level", "operator": "<", "value": 20}
```

### 3.3 数据库Schema

```sql
-- K2: 治理规则表
CREATE TABLE governance_rules (
    rule_id VARCHAR(64) PRIMARY KEY,
    rule_name VARCHAR(256) NOT NULL,
    description TEXT,
    rule_type VARCHAR(32) NOT NULL,       -- constraint | preference | policy
    scope VARCHAR(32) NOT NULL,           -- agent | zone | building | system
    priority INTEGER DEFAULT 50,
    
    -- 条件和动作
    condition JSONB NOT NULL,
    action_type VARCHAR(32) NOT NULL,     -- block | warn | modify | log | escalate
    action_config JSONB DEFAULT '{}',
    
    -- 适用范围
    applicable_agent_types TEXT[],
    applicable_building_ids TEXT[],
    applicable_zone_ids TEXT[],
    
    -- 生命周期
    effective_from TIMESTAMPTZ DEFAULT NOW(),
    effective_until TIMESTAMPTZ,
    enabled BOOLEAN DEFAULT TRUE,
    
    -- 审计
    created_by VARCHAR(64),
    created_at TIMESTAMPTZ DEFAULT NOW(),
    approved_by VARCHAR(64),
    change_log JSONB DEFAULT '[]',
    
    -- 层级继承（阶段二）
    parent_rule_id VARCHAR(64),
    override_allowed BOOLEAN DEFAULT TRUE,
    
    -- 统计
    trigger_count INTEGER DEFAULT 0,
    block_count INTEGER DEFAULT 0,
    last_triggered_at TIMESTAMPTZ
);

CREATE INDEX idx_gr_scope ON governance_rules(scope);
CREATE INDEX idx_gr_enabled ON governance_rules(enabled, effective_from, effective_until);
CREATE INDEX idx_gr_type ON governance_rules(rule_type);
```
